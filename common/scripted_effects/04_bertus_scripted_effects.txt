	if = {
		limit = {
			has_global_modifier_value = { which = monthly_military_reform_power_gain value = 0.01 }
		}
		export_to_variable = { which = monthly_military_reform_power_gain_var value = modifier:monthly_military_reform_power_gain }
		export_to_variable = { which = monthly_military_reform_multiplier_var value = modifier:military_reform_power_gain_modifier }
		multiply_variable = { which = monthly_military_reform_power_gain_var which = monthly_military_reform_multiplier_var }
		change_variable = { which = military_reform_progress_var which = monthly_military_reform_power_gain_var }
		if = {
			limit = { check_variable = { which = military_reform_progress_var value = 100 } }
			change_variable = { which = military_reforms_tokens_var value = 1 }
			change_variable = { which = military_reform_progress_var value = -100 }
		}
	}
}
get_monthly_manpower = {
	if = {
		limit = {
			NOT = { manpower_percentage = 1.0 }
			has_global_modifier_value = { which = monthly_manpower_modifier value = 0.01 }
		}

		set_variable = { which = temp value = 0 }
		export_to_variable = { which = temp value = modifier:manpower_recovery_speed }
		change_variable = { which = temp value = 1 }											# (1+x)

		set_variable = { which = manpower_recovery_var value = 0 }
		change_variable = { which = manpower_recovery_var which = temp }						#save for later

		export_to_variable = { which = max_manpower value = trigger_value:max_manpower }
		multiply_variable = { which = temp which = max_manpower }
		divide_variable = { which = temp value = 24 }
		
		set_variable = { which = monthly_manpower_gain value = 0 }
		change_variable = { which = monthly_manpower_gain which = temp }						#save for later

		export_to_variable = { which = monthly_manpower_modifier_var value = modifier:monthly_manpower_modifier }
		multiply_variable = { which = temp which = monthly_manpower_modifier_var }

		set_variable = { which = monthly_manpower_gain_after_modifiers value = 0 }
		change_variable = { which = monthly_manpower_gain_after_modifiers which = temp }		#save for later
		multiply_variable = { which = monthly_manpower_gain_after_modifiers value = 0.1 }		#save for later

		while = {
			limit = { check_variable = { which = temp value = 0 } }
			change_variable = { which = temp value = -1 }
			add_manpower = 0.1
		}
	}
}
get_monthly_sailors = {
	if = {
		limit = {
			NOT = { sailors_percentage = 1.0 }
		}
		if = {
			limit = {
				has_global_modifier_value = { which = monthly_sailors_modifier value = 0.01 }
			}

			set_variable = { which = temp value = 0 }
			export_to_variable = { which = temp value = modifier:sailors_recovery_speed }
			change_variable = { which = temp value = 1 }											# (1+x)

			set_variable = { which = sailors_recovery_var value = 0 }
			change_variable = { which = sailors_recovery_var which = temp }						#save for later

			export_to_variable = { which = max_sailors_var value = trigger_value:max_sailors }
			multiply_variable = { which = temp which = max_sailors_var }
			divide_variable = { which = temp value = 24 }
			
			set_variable = { which = monthly_sailors_gain value = 0 }
			change_variable = { which = monthly_sailors_gain which = temp }						#save for later

			export_to_variable = { which = monthly_sailors_modifier_var value = modifier:monthly_sailors_modifier }
			multiply_variable = { which = temp which = monthly_sailors_modifier_var }

			set_variable = { which = monthly_sailors_gain_after_modifiers value = 0 }
			change_variable = { which = monthly_sailors_gain_after_modifiers which = temp }		#save for later
			multiply_variable = { which = monthly_sailors_gain_after_modifiers value = 0.1 }		#save for later

			while = {
				limit = { check_variable = { which = temp value = 0 } }
				change_variable = { which = temp value = -1 }
				add_sailors = 100
			}
		}
		if = {
			limit = { has_government_attribute = manpower_to_sailors_fraction }
	
			set_variable = { which = temp value = 0 }
			change_variable = { which = temp which = monthly_manpower_gain }
			change_variable = { which = monthly_manpower_modifier_var value = 1 }					# (1+x)
			multiply_variable = { which = temp which = monthly_manpower_modifier_var }
			change_variable = { which = monthly_manpower_modifier_var value = -1 }					#back to base
			multiply_variable = { which = temp value = 0.20 }
			set_variable = { which = monthly_sailors_gain_after_modifiers value = 0 }
			change_variable = { which = monthly_sailors_gain_after_modifiers which = temp }		#save for later
			multiply_variable = { which = monthly_sailors_gain_after_modifiers value = 0.1 }		#save for later
			while = {
				limit = { check_variable = { which = temp value = 0 } }
				change_variable = { which = temp value = -1 }
				add_sailors = 100
			}
		}
	}
}

transfer_manpower_fraction = {
	$giver$ = {
		set_variable = { which = max_manpower_var value = 0 }
		set_variable = { which = manpower_to_transfer value = 1 }
		export_to_variable = { which = max_manpower_var value = max_manpower }
		subtract_variable = { which = manpower_to_transfer value = $fraction$ }
		multiply_variable = { which = manpower_to_transfer which = max_manpower_var }
		while = {
			limit = { check_variable = { which = max_manpower_var which = manpower_to_transfer } }
			subtract_variable = { which = max_manpower_var value = 0.1 }
			add_manpower = -0.1
			$receiver$ = {add_manpower = 0.1}
		}
	}
}

		}
	}

get_boosted_bonus = {
	if = {
		limit = {
			OR = {
				has_country_modifier = boosted_country_$modifier$_2
				has_country_modifier = boosted_country_$modifier$_3
			}
		}
		remove_country_modifier = boosted_country_$modifier$_2
		remove_country_modifier = boosted_country_$modifier$_3
		add_country_modifier = {
			name = boosted_country_$modifier$_3
			duration = 3650
		}
	}
	else_if = {
		limit = {
			has_country_modifier = boosted_country_$modifier$
		}
		remove_country_modifier = boosted_country_$modifier$
		add_country_modifier = {
			name = boosted_country_$modifier$_2
			duration = 3650
		}
	}
	else = {
		add_country_modifier = {
			name = boosted_country_$modifier$
			duration = 3650
		}
	}
}
get_treasury_fraction = {
	FROM = {
		set_variable = { which = current_treasury value = 0 }
		set_variable = { which = treasury_to_transfer value = 1 }
		subtract_variable = { which = treasury_to_transfer value = $fraction$ }
		export_to_variable = { which = current_treasury value = treasury }
		multiply_variable = { which = treasury_to_transfer which = current_treasury }
		while = {
			limit = { check_variable = { which = current_treasury which = treasury_to_transfer } }
			subtract_variable = { which = current_treasury value = 1 }
			add_treasury = -1
			ROOT = {add_treasury = 1}
		}
	}
}
